# test

cat hg38.fa | grep ">" > chr_name_ucsc.txt

awk '{FS="\t";OFS="\t";if(FNR != 1){print $1,$2,$3;}}' ./PeakCalls/Cancer_Type-specific_PeakCalls/ACC_peakCalls.bed | head > test.bed
# extract sequences after removing column names
